<!-- RESEARCH Pattern 5 — Six-position arithmetic + Pitfall 9 near-cursor clamp: -->
```lua
local HUD_W, HUD_H = 140, 36       -- already declared in Plan 01; do not redeclare
local EDGE_MARGIN = 24
local TOP_GAP = 50                  -- already HUD_TOP_GAP from Plan 01
local NEAR_CURSOR_OFFSET = 30

local function positionFor(name, screenFrame)
  -- screenFrame: hs.screen:fullFrame() — table with x, y, w, h in global coords.
  if name == "top-center" then
    return { x = screenFrame.x + (screenFrame.w - HUD_W) / 2, y = screenFrame.y + TOP_GAP }
  elseif name == "top-right" then
    return { x = screenFrame.x + screenFrame.w - HUD_W - EDGE_MARGIN, y = screenFrame.y + TOP_GAP }
  elseif name == "bottom-center" then
    return { x = screenFrame.x + (screenFrame.w - HUD_W) / 2, y = screenFrame.y + screenFrame.h - HUD_H - EDGE_MARGIN }
  elseif name == "bottom-right" then
    return { x = screenFrame.x + screenFrame.w - HUD_W - EDGE_MARGIN, y = screenFrame.y + screenFrame.h - HUD_H - EDGE_MARGIN }
  elseif name == "near-cursor" then
    -- Use cursor's screen, NOT focused-window screen (RESEARCH Priority 6 note)
    local mp = hs.mouse.absolutePosition()
    local cursorScreen = hs.screen.find(mp) or hs.screen.mainScreen()
    local cf = cursorScreen:fullFrame()
    -- Clamp to fit within cursor's screen (RESEARCH Pitfall 9)
    local x = math.max(cf.x + EDGE_MARGIN, math.min(mp.x - HUD_W / 2, cf.x + cf.w - HUD_W - EDGE_MARGIN))
    local y = math.max(cf.y + EDGE_MARGIN, math.min(mp.y + NEAR_CURSOR_OFFSET, cf.y + cf.h - HUD_H - EDGE_MARGIN))
    return { x = x, y = y }
  elseif name == "center" then
    return { x = screenFrame.x + (screenFrame.w - HUD_W) / 2, y = screenFrame.y + (screenFrame.h - HUD_H) / 2 }
  else
    -- Defensive default — should never hit because validPositions gates at module load.
    return { x = screenFrame.x + (screenFrame.w - HUD_W) / 2, y = screenFrame.y + TOP_GAP }
  end
end
```
