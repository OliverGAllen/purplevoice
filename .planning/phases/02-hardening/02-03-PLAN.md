hs.notify.register(tag, fn)
  - Registers a callback by tag. Survives Hammerspoon reload (per issue #1414).
  - This plan registers TWO: voiceccOpenMicSettings, voiceccOpenAccessibilitySettings.
  - Callback signature: function(notification) end

hs.notify.new([tag], attributes) -> notify
  - tag: optional string — references a previously-registered callback.
  - attributes table:
      title              = string (bold)
      informativeText    = string (body)
      actionButtonTitle  = string ("Open Settings")
      hasActionButton    = boolean (true to show button — Alerts mode only)
      autoWithdraw       = boolean (false = persist)
      withdrawAfter      = number (0 = forever; otherwise seconds)
  - CRITICAL CAVEAT: actionButtonTitle/hasActionButton only have visible effect
    when user has set Hammerspoon's notification style to "Alerts" in
    System Settings → Notifications → Hammerspoon. Banner mode shows the
    notification but no button. The notification BODY must therefore be
    self-explanatory, not dependent on the button.

notify:send()
  - Dispatches the notification.

hs.urlevent.openURL(url)
  - Opens a URL via the default handler. For System Settings, use the
    x-apple.systempreferences: scheme.

  local lastNotifyAt = {}   -- exit_code -> timestamp seconds

Helper:
  local function notifyOnce(exitCode, factory)
    local now = hs.timer.absoluteTime() / 1e9
    local last = lastNotifyAt[exitCode] or 0
    if (now - last) < NOTIFY_COOLDOWN_S then return end
    lastNotifyAt[exitCode] = now
