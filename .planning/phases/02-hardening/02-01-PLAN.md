```bash
#!/usr/bin/env bash
# voice-cc-record — Phase 2 hardened bash glue
# Owns sox capture + transcription pipeline. Pattern 2 boundary: WHISPER_BIN
# appears exactly twice (assignment + single use inside transcribe()).
#
# Phase 2 additions over Phase 1:
#   - VAD flags + Silero model in transcribe() body (TRA-04)
#   - Duration gate via soxi -D (TRA-05)
#   - Denylist exact-match filter (TRA-06)
#   - Empty/whitespace transcript drop (INJ-04)
#   - TCC stderr fingerprint detection (ROB-02)
#   - EXIT trap covering ALL exit paths including SIGINT (ROB-04)
#   - Semantic exit codes 2/3/10/11/12 for Lua dispatch
#   - --no-prints + removal of -otxt -of (Phase-1 TODO c — suppress sibling .txt)
#   - VOICE_CC_TEST_SKIP_SOX env var hook for tests/test_*.sh

set -euo pipefail

# --- Absolute binary paths (Pitfall 2, ROB-03) -------------------------
SOX_BIN="${SOX_BIN:-/opt/homebrew/bin/sox}"
SOXI_BIN="${SOXI_BIN:-/opt/homebrew/bin/soxi}"
WHISPER_BIN="${WHISPER_BIN:-/opt/homebrew/bin/whisper-cli}"

# --- Paths (D-03 XDG layout) -------------------------------------------
MODEL="${VOICE_CC_MODEL:-$HOME/.local/share/voice-cc/models/ggml-small.en.bin}"
SILERO_MODEL="${VOICE_CC_SILERO_MODEL:-$HOME/.local/share/voice-cc/models/ggml-silero-v6.2.0.bin}"
VOCAB_FILE="$HOME/.config/voice-cc/vocab.txt"
DENYLIST="$HOME/.config/voice-cc/denylist.txt"
VAD_THRESHOLD="${VOICE_CC_VAD_THRESHOLD:-0.50}"
WAV_DIR="/tmp/voice-cc"
WAV="$WAV_DIR/recording.wav"
SOX_ERR_LOG="$WAV_DIR/sox.stderr"

mkdir -p "$WAV_DIR"

# --- Per-startup sweep of stale leftovers (>5 min old) -----------------
# Defends against SIGKILL exit paths (which don't fire EXIT trap).
find "$WAV_DIR" -maxdepth 1 -name "*.wav" -mmin +5 -delete 2>/dev/null || true
find "$WAV_DIR" -maxdepth 1 -name "*.txt" -mmin +5 -delete 2>/dev/null || true
find "$WAV_DIR" -maxdepth 1 -name "sox.stderr" -mmin +5 -delete 2>/dev/null || true

# --- EXIT trap covers ALL exit paths (success, exit codes, signals) ----
# Bash's EXIT trap fires on every exit including after signal-specific traps.
trap 'rm -f "$WAV" "$SOX_ERR_LOG"' EXIT

# --- Verify required files exist (TRA-04, TRA-01) ----------------------
if [ ! -x "$WHISPER_BIN" ]; then
  echo "voice-cc: whisper-cli missing at $WHISPER_BIN — run setup.sh" >&2
  exit 11
fi
if [ ! -f "$MODEL" ]; then
  echo "voice-cc: Whisper model missing at $MODEL — run setup.sh" >&2
  exit 11
fi
if [ ! -f "$SILERO_MODEL" ]; then
  echo "voice-cc: Silero VAD weights missing at $SILERO_MODEL — run setup.sh" >&2
  exit 11
fi

# --- Read vocab if available (TRA-03) ----------------------------------
if [ -r "$VOCAB_FILE" ]; then
  VOCAB="$(cat "$VOCAB_FILE")"
else
  VOCAB=""
fi

# --- transcribe() — the SOLE STT abstraction boundary (Pattern 2) ------
# v1: one-shot whisper-cli with VAD + suppress-nst.
# v1.1 (deferred): swap body to `curl 127.0.0.1:8080/inference -F file=@"$1"`.
# NOTHING else in this script may invoke whisper-cli directly.
transcribe() {
  local wav_path="$1"
  "$WHISPER_BIN" \
    -m "$MODEL" \
    --language en \
    --no-timestamps \
    --no-prints \
    --vad \
    --vad-model "$SILERO_MODEL" \
    --vad-threshold "$VAD_THRESHOLD" \
    --suppress-nst \
    --prompt "$VOCAB" \
    -f "$wav_path" 2>/dev/null
}

# --- Capture: sox in background, signals forwarded for clean WAV finalise -
# VOICE_CC_TEST_SKIP_SOX=1 → skip capture (tests pre-stage the WAV).
if [ "${VOICE_CC_TEST_SKIP_SOX:-0}" != "1" ]; then
  : > "$SOX_ERR_LOG"   # initialise empty (so grep doesn't error)
  "$SOX_BIN" -d -r 16000 -c 1 -b 16 "$WAV" >/dev/null 2>"$SOX_ERR_LOG" &
  SOX_PID=$!
  trap 'kill -TERM "$SOX_PID" 2>/dev/null || true' TERM INT
  # CRITICAL: capture wait's exit status correctly. The pattern
  #   wait "$SOX_PID" || true
  #   SOX_EXIT=$?
  # is BROKEN — `|| true` runs after wait fails, returns 0, then $? is 0
  # always. SOX_EXIT would never be non-zero, breaking TCC detection below.
  # END-STATE pattern (use this verbatim):
  SOX_EXIT=0
  wait "$SOX_PID" || SOX_EXIT=$?
  trap - TERM INT

  # --- TCC silent-deny detection (ROB-02) ------------------------------
  if [ "$SOX_EXIT" -ne 0 ]; then
    if grep -qE 'Permission denied|AudioObject(GetPropertyData|SetPropertyData)|kAudio.*Error' "$SOX_ERR_LOG" 2>/dev/null; then
      exit 10
    fi
    exit 12
  fi
fi

# --- Duration gate (TRA-05, success criterion #1) ----------------------
DURATION=$("$SOXI_BIN" -D "$WAV" 2>/dev/null || echo 0)
if awk -v d="$DURATION" 'BEGIN { exit !(d < 0.4) }'; then
  exit 2
fi

# --- Transcribe (Pattern 2 boundary — sole whisper-cli call) -----------
TRANSCRIPT="$(transcribe "$WAV")"
TRANSCRIPT="$(printf "%s" "$TRANSCRIPT" | sed -e 's/^[[:space:]]*//' -e 's/[[:space:]]*$//')"

# --- Empty drop (INJ-04) -----------------------------------------------
if [ -z "$TRANSCRIPT" ]; then
  exit 3
fi

# --- Denylist exact-match (TRA-06) -------------------------------------
if [ -r "$DENYLIST" ]; then
  CANON_T="$(printf "%s" "$TRANSCRIPT" | tr -d '[:space:]' | tr '[:upper:]' '[:lower:]')"
  while IFS= read -r phrase; do
    case "$phrase" in ''|'#'*) continue ;; esac
    CANON_P="$(printf "%s" "$phrase" | tr -d '[:space:]' | tr '[:upper:]' '[:lower:]')"
    [ -z "$CANON_P" ] && continue
    if [ "$CANON_T" = "$CANON_P" ]; then
      exit 3
    fi
  done < "$DENYLIST"
fi

# --- Emit transcript (no trailing newline — caller pastes verbatim) ----
printf "%s" "$TRANSCRIPT"
```
